870 static void aiGame::registerGame(const std::string& gameName) {
871
872 if (aiGames_.find(gameName) == aiGames_.cend()) {
873
874 T* newGame = new T();
875
876 aiGames_.insert({ gameName, newGame });
877 gamesRegisterOrder_.push_back(newGame);
878
879 // Create the directories for the records and AI data for these games
880 // if not already created.
881 std::string recordsPath = "records/" + gameName,
882 AIDataPath = "AIData/" + gameName,
883 terrainPath = "saves/recordingWorlds/" + gameName;
884 if (std::filesystem::exists(recordsPath)) {
885
886 if (!std::filesystem::is_directory(recordsPath))
887 logger::errorLog(recordsPath + " is occupied by a file. Please move it.");
888
889 }
890 else
891 std::filesystem::create_directory(recordsPath);
892
893 if (std::derived_from<T, trainingGame>) {
894
895 if (std::filesystem::exists(AIDataPath)) {
896
897 if (!std::filesystem::is_directory(AIDataPath))
898 logger::errorLog(AIDataPath + " is occupied by a file. Please move it.");
899
900 }
901 else
902 std::filesystem::create_directory(AIDataPath);
903
904 }
905
906 if (std::filesystem::exists(terrainPath)) {
907
908 if (!std::filesystem::is_directory(terrainPath))
909 logger::errorLog(terrainPath + " is occupied by a file. Please move it.");
910
911 }
912 else
913 std::filesystem::create_directory("saves/recordingWorlds/" + gameName);
914
915 }
916 else
917 logger::errorLog("An AI game with the name " + gameName + " is already registered");
918
919 }
